while True:
    lines = int(input("Escribe un número válido: "))
    if(lines % 2 == 0):
        print("El número introducido es par, introduce uno impar")
    else:
        print("Correcto, número impar")
        for numlines in range(1,lines+1):
            spaces = lines - numlines
            num = 2 * numlines - 1
            print(" " * spaces + '*' * num)
